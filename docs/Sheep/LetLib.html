def record_kv_by_id(pair, datastore, store_id, key_id, value_id)
  value = pair[value_id]
  key = pair[key_id]
  ldebug "store => '#{store_id}', key_id => '#{key_id}', value_id => '#{value_id}', "\
    "pair_data => '#{pair}', key_id => '#{key_id}', value_id => '#{value_id}', "\
    "key => '#{key}', value => '#{value}'"
  datastore.assign(store_id, value, key)
end
